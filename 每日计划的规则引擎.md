// constraints.ts

import { UserProfile, UserState, Constraint } from './types'

export function generateConstraints(
  user: UserProfile,
  state: UserState,
  target: { calories: number; protein: number }
): Constraint {
  const includeTags: string[] = []
  const excludeTags: string[] = []

  // 🎯 目标驱动
  if (user.goal === 'fat_loss') {
    includeTags.push('高蛋白', '低热量', '高饱腹')
  }

  if (user.goal === 'muscle_gain') {
    includeTags.push('高蛋白', '高碳水')
  }

  if (user.goal === 'health') {
    includeTags.push('均衡', '天然')
  }

  // 🎯 状态驱动（关键）
  const proteinGap =
    user.targetProtein - state.consumed.protein

  const calorieGap =
    user.targetCalories - state.consumed.calories

  if (proteinGap > 30) {
    includeTags.push('高蛋白')
  }

  if (calorieGap < 300) {
    includeTags.push('低热量')
  }

  if (calorieGap < 0) {
    includeTags.push('超低热量')
    excludeTags.push('高脂肪')
  }

  // 🎯 场景
  if (user.scenario === 'takeout') {
    includeTags.push('外卖')
  }

  // 🎯 用户偏好
  user.dislikes?.forEach(tag => excludeTags.push(tag))

  return {
    includeTags,
    excludeTags,
    maxCalories: target.calories * 1.1,
    minProtein: target.protein * 0.6,
  }
}

// filter.ts

import { Food, Constraint } from './types'

export function filterFoods(
  foods: Food[],
  constraint: Constraint
) {
  return foods.filter(food => {
    // 标签包含
    if (
      constraint.includeTags &&
      !constraint.includeTags.every(tag =>
        food.tags.includes(tag)
      )
    ) {
      return false
    }

    // 标签排除
    if (
      constraint.excludeTags &&
      constraint.excludeTags.some(tag =>
        food.tags.includes(tag)
      )
    ) {
      return false
    }

    // 热量限制
    if (
      constraint.maxCalories &&
      food.calories > constraint.maxCalories
    ) {
      return false
    }

    // 蛋白质
    if (
      constraint.minProtein &&
      food.protein < constraint.minProtein
    ) {
      return false
    }

    return true
  })
}

// score.ts

import { Food, Goal } from './types'

export function scoreFood(food: Food, goal: Goal) {
  const normalize = (v: number, max: number) =>
    Math.min(v / max, 1)

  const caloriesScore = 1 - normalize(food.calories, 800)
  const proteinScore = normalize(food.protein, 50)
  const carbsScore = 1 - normalize(food.carbs, 100)
  const fatScore = 1 - normalize(food.fat, 50)
  const qualityScore = food.quality / 10
  const satietyScore = food.satiety / 10

  const weights = {
    fat_loss: [0.3, 0.25, 0.15, 0.1, 0.1, 0.1],
    muscle_gain: [0.25, 0.3, 0.2, 0.1, 0.1, 0.05],
    health: [0.15, 0.1, 0.1, 0.1, 0.3, 0.15],
    habit: [0.2, 0.15, 0.1, 0.1, 0.25, 0.2],
  }[goal]

  const scores = [
    caloriesScore,
    proteinScore,
    carbsScore,
    fatScore,
    qualityScore,
    satietyScore,
  ]

  return scores.reduce(
    (sum, s, i) => sum + s * weights[i],
    0
  )
}

// diversify.ts

import { Food } from './types'

export function diversify(
  foods: Food[],
  lastFoods: string[] = []
) {
  return foods
    .filter(f => !lastFoods.includes(f.id))
    .slice(0, 3)
}

// engine.ts

import {
  UserProfile,
  UserState,
  Food,
} from './types'

import { calcNextMealTarget } from './target'
import { generateConstraints } from './constraints'
import { filterFoods } from './filter'
import { scoreFood } from './score'
import { diversify } from './diversify'

export function recommendNextMeal(
  user: UserProfile,
  state: UserState,
  foods: Food[]
) {
  const target = calcNextMealTarget(user, state)

  const constraints = generateConstraints(
    user,
    state,
    target
  )

  const candidates = filterFoods(foods, constraints)

  const scored = candidates
    .map(f => ({
      food: f,
      score: scoreFood(f, user.goal),
    }))
    .sort((a, b) => b.score - a.score)

  return diversify(
    scored.map(s => s.food),
    state.lastFoods
  )
}
