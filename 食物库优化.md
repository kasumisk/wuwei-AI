👉 改造 diversify

export function diversify(foods: Food[]) {
  const usedCategories = new Set()
  const result: Food[] = []

  for (const food of foods) {
    if (!usedCategories.has(food.category)) {
      result.push(food)
      usedCategories.add(food.category)
    }

    if (result.length >= 3) break
  }

  return result
}

function similarity(a: Food, b: Food) {
  let score = 0

  if (a.category === b.category) score += 0.5

  const commonTags = a.tags.filter(tag =>
    b.tags.includes(tag)
  )

  score += commonTags.length * 0.1

  return score
}

👉 改造排序（核心）

export function diversifyWithPenalty(
  foods: { food: Food; score: number }[]
) {
  const result: Food[] = []

  while (result.length < 3 && foods.length > 0) {
    let bestIndex = 0
    let bestScore = -Infinity

    foods.forEach((item, i) => {
      let penalty = 0

      for (const selected of result) {
        penalty += similarity(item.food, selected)
      }

      const finalScore = item.score - penalty

      if (finalScore > bestScore) {
        bestScore = finalScore
        bestIndex = i
      }
    })

    result.push(foods[bestIndex].food)
    foods.splice(bestIndex, 1)
  }

  return result
}

👉 ✅ 效果：

第一名：鸡胸肉
第二名：鸡蛋（被惩罚）
第三名：沙拉（胜出）

function addExploration(foods: Food[]) {
  const top = foods.slice(0, 5)

  return top.sort(() => Math.random() - 0.5).slice(0, 3)
}
