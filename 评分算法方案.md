🧮 5. 各维度评分算法

⸻

① 热量（Energy Score）

⸻


energyScore = clamp(
  100 - (|actual - target| / target) * 100,
  0,
  100
)


⸻

👉 说明：

偏差	分数
±5%	≈ 95
±20%	≈ 80
>50%	低分


⸻

⸻

② 蛋白质占比（Protein Ratio）

⸻


proteinRatio = proteinCalories / totalCalories


⸻

目标范围：

目标	范围
减脂	25%~35%
增肌	25%~40%
健康	15%~25%


⸻


score = rangeScore(proteinRatio, min, max)


⸻

⸻

③ 宏量结构（Macro Balance）

⸻

理想比例：

carbs: 40~55%
fat: 20~30%


⸻


macroScore = average(
  rangeScore(carbsRatio, carbMin, carbMax),
  rangeScore(fatRatio, fatMin, fatMax)
)


⸻

⸻

④ 食物质量（Food Quality）

⸻

来源：

AI识别 / 食物数据库 / 加工度判断


⸻


score = quality * 10 // 1~10 → 10~100


⸻

👉 示例：

食物	分数
炸鸡	30
沙拉	85
全食物	90+


⸻

⸻

⑤ 饱腹感（Satiety）

⸻

来源：

用户反馈 or AI预测（蛋白+纤维）


⸻


score = satiety * 10


⸻

⸻

⑥ 稳定性（Stability，可选）

⸻


基于：
- 连续记录天数
- 餐次规律

🧠 二、完整 TypeScript 实现（可直接用）

⸻


type NutritionInput = {
  calories: number
  targetCalories: number
  protein: number
  carbs: number
  fat: number
  foodQuality: number // 1~10
  satiety: number // 1~10
}

function clamp(v: number, min = 0, max = 100) {
  return Math.max(min, Math.min(max, v))
}

function rangeScore(value: number, min: number, max: number) {
  if (value >= min && value <= max) return 100
  const diff = value < min ? min - value : value - max
  return clamp(100 - diff * 200) // 偏差越大扣分越多
}

function calcEnergyScore(actual: number, target: number) {
  return clamp(100 - (Math.abs(actual - target) / target) * 100)
}

function calcProteinRatioScore(
  protein: number,
  calories: number,
  goal: GoalType
) {
  const proteinCalories = protein * 4
  const ratio = proteinCalories / calories

  const ranges = {
    fat_loss: [0.25, 0.35],
    muscle_gain: [0.25, 0.4],
    health: [0.15, 0.25],
    habit: [0.15, 0.3],
  }

  const [min, max] = ranges[goal]
  return rangeScore(ratio, min, max)
}

function calcMacroScore(carbs: number, fat: number, calories: number) {
  const carbRatio = (carbs * 4) / calories
  const fatRatio = (fat * 9) / calories

  const carbScore = rangeScore(carbRatio, 0.4, 0.55)
  const fatScore = rangeScore(fatRatio, 0.2, 0.3)

  return (carbScore + fatScore) / 2
}

export function calculateNutritionScore(
  input: NutritionInput,
  goal: GoalType
) {
  const energy = calcEnergyScore(
    input.calories,
    input.targetCalories
  )

  const proteinRatio = calcProteinRatioScore(
    input.protein,
    input.calories,
    goal
  )

  const macroBalance = calcMacroScore(
    input.carbs,
    input.fat,
    input.calories
  )

  const foodQuality = input.foodQuality * 10
  const satiety = input.satiety * 10

  const weights = WEIGHTS[goal]

  let score =
    energy * weights.energy +
    proteinRatio * weights.proteinRatio +
    macroBalance * weights.macroBalance +
    foodQuality * weights.foodQuality +
    satiety * weights.satiety

  // 惩罚机制
  if (input.calories > input.targetCalories * 1.3) {
    score *= 0.7
  }

  return Math.round(score)
}
